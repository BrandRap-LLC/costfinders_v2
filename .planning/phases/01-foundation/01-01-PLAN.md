3. Create biome.json with sensible defaults:

       "enabled": true,
       "rules": { "recommended": true }

     "formatter": {
       "enabled": true,
       "indentStyle": "space",
       "indentWidth": 2
     },
     "javascript": {
       "formatter": { "quoteStyle": "single", "semicolons": "asNeeded" }
     }
   }
