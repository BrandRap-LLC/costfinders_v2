```tsx
'use client'

import { Syringe, Drop, Sparkle, Sun } from '@phosphor-icons/react'
import type { TreatmentCategory } from '@/types'

interface CategoryFilterProps {
  selected: TreatmentCategory | 'all'
  onChange: (category: TreatmentCategory | 'all') => void
}

type CategoryOption = {
  value: TreatmentCategory | 'all'
  label: string
  icon?: React.ElementType
}

const categories: CategoryOption[] = [
  { value: 'all', label: 'All Deals' },
  { value: 'botox', label: 'Botox', icon: Syringe },
  { value: 'fillers', label: 'Fillers', icon: Drop },
  { value: 'facials', label: 'Facials', icon: Sparkle },
  { value: 'laser', label: 'Laser', icon: Sun },
]

export function CategoryFilter({ selected, onChange }: CategoryFilterProps) {
  return (
    <div className="flex flex-wrap gap-2">
      {categories.map((cat) => {
        const isSelected = selected === cat.value
        const Icon = cat.icon

        return (
          <button
            key={cat.value}
            onClick={() => onChange(cat.value)}
            className={`
              inline-flex items-center gap-1.5 px-4 py-2
              rounded-full text-sm font-medium
              transition-all duration-200
              ${isSelected
                ? 'bg-brand-primary text-white shadow-md'
                : 'bg-glass-bg border border-glass-border text-text-secondary hover:text-text-primary hover:border-glass-border-hover'
              }
            `}
          >
            {Icon && <Icon size={16} weight={isSelected ? 'fill' : 'regular'} />}
            {cat.label}
          </button>
        )
      })}
    </div>
  )
}
```

Key implementation notes:
- Chip-style buttons with icon + label
- Selected state uses brand-primary background
- Unselected uses glassmorphic styling
- Icons are filled when selected, regular when not
- Flexible layout with wrap for mobile

```tsx
'use client'

import { useState } from 'react'
import { SquaresFour, List, MagnifyingGlass } from '@phosphor-icons/react'
import { DealCard } from './dealCard'
import { DealCardSkeleton } from './dealCardSkeleton'
import type { AnonymousDeal } from '@/types'

type ViewMode = 'grid' | 'list'

interface DealsGridProps {
  deals: AnonymousDeal[]
  isLoading?: boolean
  onDealClick?: (dealId: string) => void
}

export function DealsGrid({ deals, isLoading = false, onDealClick }: DealsGridProps) {
  const [viewMode, setViewMode] = useState<ViewMode>('grid')

  // Loading state
  if (isLoading) {
    return (
      <div>
        <ViewModeToggle mode={viewMode} onChange={setViewMode} disabled />
        <div className={viewMode === 'grid'
          ? 'grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 mt-4'
          : 'flex flex-col gap-4 mt-4'
        }>
          {Array.from({ length: 6 }).map((_, i) => (
            <DealCardSkeleton key={i} variant={viewMode} />
          ))}
        </div>
      </div>
    )
  }

  // Empty state
  if (deals.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-16 text-center">
        <div className="w-16 h-16 rounded-full bg-glass-bg flex items-center justify-center mb-4">
          <MagnifyingGlass size={32} weight="light" className="text-text-muted" />
        </div>
        <h3 className="text-lg font-semibold text-text-primary mb-2">
          No deals found
        </h3>
        <p className="text-text-secondary max-w-sm">
          Try adjusting your filters or check back later for new deals in your area.
        </p>
      </div>
    )
  }

  return (
    <div>
      <div className="flex items-center justify-between mb-4">
        <p className="text-sm text-text-secondary">
          {deals.length} {deals.length === 1 ? 'deal' : 'deals'} found
        </p>
        <ViewModeToggle mode={viewMode} onChange={setViewMode} />
      </div>

      <div className={viewMode === 'grid'
        ? 'grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4'
        : 'flex flex-col gap-4'
      }>
        {deals.map((deal) => (
          <DealCard
            key={deal.id}
            deal={deal}
            variant={viewMode}
            onClick={() => onDealClick?.(deal.id)}
          />
        ))}
      </div>
    </div>
  )
}

// View mode toggle component
interface ViewModeToggleProps {
  mode: ViewMode
  onChange: (mode: ViewMode) => void
  disabled?: boolean
}

function ViewModeToggle({ mode, onChange, disabled }: ViewModeToggleProps) {
  return (
    <div className="flex items-center gap-1 p-1 bg-glass-bg rounded-lg border border-glass-border">
      <button
        onClick={() => onChange('grid')}
        disabled={disabled}
        className={`
          p-1.5 rounded-md transition-colors
          ${mode === 'grid'
            ? 'bg-brand-primary/20 text-brand-primary'
            : 'text-text-tertiary hover:text-text-secondary'
          }
          disabled:opacity-50 disabled:cursor-not-allowed
        `}
        aria-label="Grid view"
      >
        <SquaresFour size={18} weight={mode === 'grid' ? 'fill' : 'regular'} />
      </button>
      <button
        onClick={() => onChange('list')}
        disabled={disabled}
        className={`
          p-1.5 rounded-md transition-colors
          ${mode === 'list'
            ? 'bg-brand-primary/20 text-brand-primary'
            : 'text-text-tertiary hover:text-text-secondary'
          }
          disabled:opacity-50 disabled:cursor-not-allowed
        `}
        aria-label="List view"
      >
        <List size={18} weight={mode === 'list' ? 'fill' : 'regular'} />
      </button>
    </div>
  )
}
```

Key implementation notes:

- View mode toggle with Phosphor icons
- Loading state shows 6 skeletons
- Empty state with icon and helpful message
- Deal count displayed above grid

```tsx
'use client'

import { useState, useMemo } from 'react'
import { useRouter } from 'next/navigation'
import { MapPin } from '@phosphor-icons/react'
import { CategoryFilter } from '@/components/patterns/categoryFilter'
import { DealsGrid } from '@/components/features/dealsGrid'
import { getActiveDeals, getDealsByCategory } from '@/lib/mock-data'
import type { TreatmentCategory } from '@/types'

export default function DealsPage() {
  const router = useRouter()
  const [selectedCategory, setSelectedCategory] = useState<TreatmentCategory | 'all'>('all')

  // TODO: In Phase 2 integration, use location context
  // const { location } = useLocation()
  const currentLocation = 'Austin, TX' // Placeholder

  const deals = useMemo(() => {
    if (selectedCategory === 'all') {
      return getActiveDeals()
    }
    return getDealsByCategory(selectedCategory)
  }, [selectedCategory])

  const handleDealClick = (dealId: string) => {
    router.push(`/deals/${dealId}`)
  }

  return (
    <main className="pt-20 pb-20 md:pb-8 px-4 sm:px-6 lg:px-8">
      <div className="max-w-7xl mx-auto">
        {/* Page Header */}
        <div className="mb-8">
          <h1 className="text-2xl sm:text-3xl font-bold text-text-primary mb-2">
            Deals Near You
          </h1>
          <div className="flex items-center gap-1.5 text-text-secondary">
            <MapPin size={16} weight="fill" />
            <span>{currentLocation}</span>
            <button className="text-brand-primary text-sm hover:underline ml-2">
              Change
            </button>
          </div>
        </div>

        {/* Category Filter */}
        <div className="mb-6">
          <CategoryFilter
            selected={selectedCategory}
            onChange={setSelectedCategory}
          />
        </div>

        {/* Deals Grid */}
        <DealsGrid
          deals={deals}
          onDealClick={handleDealClick}
        />
      </div>
    </main>
  )
}
```

- Follows standard page layout pattern from CLAUDE.md (pt-20 for header clearance)
